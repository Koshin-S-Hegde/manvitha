Ensure you are in the folder where this file is downloaded before running any commands (IMPORTANT)
You can do this with:
    cd ~
    cd project/manvitha/
    Replace "project/manvitha" with the folder this file is present in of this file

Automatic:
    sudo sh ./setup.sh

Manual:
    # Installing dependencies
    apt update
    apt install -y python3 python3-venv curl zstd
    # Installing ollama
    curl -fsSL https://ollama.com/install.sh | sh
    service ollama start
    # setting up environment
    python3 -m venv venv
    source venv/bin/activate
    pip install -r requirements-demo.txt
    ollama pull nomic-embed-text
    python3
